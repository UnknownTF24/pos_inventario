// --- REEMPLAZA TU EVENT LISTENER DEL CHECKOUT POR ESTE --- checkoutBtn.addEventListener("click", async () => { if (cart.length === 0) return; const payload = { items: cart.map(item => ({ codigo: item.codigo, cantidad: item.cantidad })) }; const totalVenta = cart.reduce((sum, item) => sum + (item.precio * item.cantidad), 0); const cartCopia = [...cart]; // Guardamos copia para el ticket try { const response = await fetch(`${API_URL}/ventas`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify(payload) }); if (response.ok) { cart = []; updateCartUI(); Swal.fire({ title: '¡Venta Exitosa!', text: `Total cobrado: Q${totalVenta.toFixed(2)}`, icon: 'success', showCancelButton: true, confirmButtonColor: '#3085d6', cancelButtonColor: '#6b7280', confirmButtonText: '🖨️ Imprimir Recibo', cancelButtonText: 'Cerrar' }).then((result) => { if (result.isConfirmed) { imprimirTicket(cartCopia, totalVenta); } }); } else { const data = await response.json(); Swal.fire({ icon: 'error', title: 'Error', text: data.detail }); } } catch (error) { Swal.fire({ icon: 'error', title: 'Fallo de Red', text: 'Error al procesar.' }); } }); // NUEVA FUNCIÓN PARA EL TICKET function imprimirTicket(items, total) { let ventana = window.open('', 'PRINT', 'height=600,width=400'); let fecha = new Date().toLocaleString(); let html = `

MI TIENDA

Recibo de Compra

${fecha}

${items.map(item => ` `).join('')}
CantArtículoSub
${item.cantidad} ${item.nombre.substring(0, 15)} Q${(item.precio * item.cantidad).toFixed(2)}
TOTAL: Q${total.toFixed(2)}
¡Gracias por su compra!
`; ventana.document.write(html); ventana.document.close(); ventana.focus(); // Pausa de medio segundo para que renderice y luego imprime setTimeout(() => { ventana.print(); ventana.close(); }, 500); }
